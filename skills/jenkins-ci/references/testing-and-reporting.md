# Testing & Reporting — JUnit Reports, Coverage, and Quality Analysis

## Overview

Jenkins provides powerful tools for capturing, analyzing, and trending test results and code quality metrics. This reference covers JUnit XML parsing, code coverage reporting (JaCoCo, Cobertura), static analysis integration (SonarQube, Checkstyle, PMD, FindBugs), and email/notification configuration.

---

## JUnit Test Reports

### Basic JUnit Parsing

The `junit` step parses JUnit XML test results and makes them available in Jenkins UI:

```groovy
pipeline {
  agent any

  stages {
    stage('Test') {
      steps {
        sh 'npm test -- --reporters junit'  // Generate test-results.xml
      }
    }
  }

  post {
    always {
      junit 'test-results.xml'  // Parse JUnit XML
    }
  }
}
```

### JUnit Step Syntax

```groovy
post {
  always {
    junit(
      testResults: '**/test-results.xml',        // File pattern
      allowEmptyResults: false,                  // Fail if no results
      keepLongStdio: true,                       // Keep full output
      healthScaleFactor: 1.0                     // Weighting for health
    )
  }
}
```

### JUnit XML Format

Standard JUnit XML structure (generated by most test frameworks):

```xml
<?xml version="1.0" encoding="UTF-8"?>
<testsuites>
  <testsuite name="unit-tests" tests="3" failures="1" skipped="0" time="1.23">
    <testcase classname="com.example.MathTest" name="testAdd" time="0.01"/>
    <testcase classname="com.example.MathTest" name="testSubtract" time="0.02">
      <failure message="Expected 5 but got 4">
        java.lang.AssertionError: Expected 5 but got 4
      </failure>
    </testcase>
    <testcase classname="com.example.MathTest" name="testMultiply" time="0.015">
      <skipped message="TODO: Implement"/>
    </testcase>
  </testsuite>
</testsuites>
```

## Archive Artifacts

Store build outputs for download or downstream jobs:

```groovy
pipeline {
  agent any

  stages {
    stage('Build') {
      steps {
        sh 'mvn clean package'
      }
    }
  }

  post {
    always {
      archiveArtifacts artifacts: 'target/*.jar', allowEmptyArchive: false
      archiveArtifacts artifacts: 'build/reports/**', fingerprint: true
    }
  }
}
```

### archiveArtifacts Options

```groovy
archiveArtifacts(
  artifacts: 'pattern1,pattern2',          // File patterns
  excludes: 'exclude-pattern',             // Patterns to exclude
  allowEmptyArchive: false,                // Fail if no files match
  onlyIfSuccessful: false,                 // Only archive if build succeeds
  fingerprint: true,                       // Track artifact versions
  defaultExcludes: true                    // Exclude SCM, .git, etc.
)
```

### Common Artifact Patterns

```groovy
// Java/Maven builds
archiveArtifacts 'target/*.jar,target/*.war'

// Python packages
archiveArtifacts 'dist/**'

// Docker images (as tarballs)
archiveArtifacts 'build/image.tar.gz'

// Reports (HTML, XML)
archiveArtifacts 'reports/**,coverage/**'

// All build outputs
archiveArtifacts 'build/**,target/**'
```

## Code Coverage Reports

### JaCoCo (Java Code Coverage)

JaCoCo is the standard code coverage tool for Java. Configure Maven or Gradle to generate coverage reports:

```groovy
pipeline {
  agent any

  stages {
    stage('Test') {
      steps {
        sh 'mvn clean test jacoco:report'  // Generate target/site/jacoco/index.html
      }
    }
  }

  post {
    always {
      publishHTML(target: [
        reportDir: 'target/site/jacoco',
        reportFiles: 'index.html',
        reportName: 'JaCoCo Coverage'
      ])
      step([$class: 'JacocoPublisher'])  // Parse coverage metrics
    }
  }
}
```

### Cobertura (Multi-Language Coverage)

Cobertura supports Java, C#, Python, and others:

```groovy
pipeline {
  agent any

  stages {
    stage('Test') {
      steps {
        sh 'mvn clean test cobertura:cobertura'  // Generate cobertura.xml
      }
    }
  }

  post {
    always {
      step([$class: 'CoberturaPublisher',
            autoUpdateHealth: false,
            autoUpdateStability: false,
            coberturaReportFile: 'target/site/cobertura/coverage.xml',
            failUnhealthy: false,
            failUnstable: false,
            maxNumberOfBuilds: 0,
            onlyStable: false,
            sourceEncoding: 'ASCII',
            zoomCoverageChart: false])
    }
  }
}
```

### OpenCover (C#/.NET)

```groovy
stages {
  stage('Test') {
    agent { label 'windows' }
    steps {
      bat '''
        nuget restore
        OpenCover.Console.exe -target:"dotnet.exe" -targetargs:"test" -output:"coverage.xml"
      '''
    }
  }
}

post {
  always {
    publishHTML(target: [
      reportDir: 'CoverageReport',
      reportFiles: 'index.html',
      reportName: 'Code Coverage'
    ])
  }
}
```

## Code Quality Analysis — SonarQube

### SonarQube Integration

SonarQube performs static analysis on code and displays results in Jenkins:

```groovy
pipeline {
  agent any

  environment {
    SONAR_HOST_URL = 'https://sonarqube.example.com'
    SONAR_LOGIN = credentials('sonarqube-token')
  }

  stages {
    stage('Build') {
      steps {
        sh 'mvn clean package'
      }
    }

    stage('SonarQube Analysis') {
      steps {
        sh '''
          mvn sonar:sonar \
            -Dsonar.projectKey=my-app \
            -Dsonar.sources=src \
            -Dsonar.host.url=${SONAR_HOST_URL} \
            -Dsonar.login=${SONAR_LOGIN}
        '''
      }
    }
  }

  post {
    always {
      junit 'target/surefire-reports/**/*.xml'
    }
  }
}
```

### SonarQube Quality Gates

Gate the pipeline based on SonarQube quality metrics:

```groovy
stages {
  stage('Quality Gate') {
    steps {
      script {
        timeout(time: 1, unit: 'HOURS') {
          waitForQualityGate abortPipeline: true  // Fail if quality gate fails
        }
      }
    }
  }
}
```

## Static Analysis Tools

### Checkstyle (Code Style)

Checkstyle checks Java code style and best practices:

```groovy
pipeline {
  agent any

  stages {
    stage('Checkstyle') {
      steps {
        sh 'mvn checkstyle:checkstyle'  // Generates target/checkstyle-result.xml
      }
    }
  }

  post {
    always {
      recordIssues(
        enabledForFailure: true,
        tools: [checkStyle(pattern: 'target/checkstyle-result.xml')]
      )
    }
  }
}
```

### PMD (Bug Detection)

PMD finds common programming mistakes:

```groovy
stages {
  stage('PMD') {
    steps {
      sh 'mvn pmd:pmd'  // Generates target/pmd.xml
    }
  }
}

post {
  always {
    recordIssues(
      enabledForFailure: true,
      tools: [pmdParser(pattern: 'target/pmd.xml')]
    )
  }
}
```

### FindBugs/SpotBugs (Bug Analysis)

Analyzes compiled Java bytecode for bugs:

```groovy
stages {
  stage('SpotBugs') {
    steps {
      sh 'mvn spotbugs:spotbugs'  // Generates target/spotbugsXml.xml
    }
  }
}

post {
  always {
    recordIssues(
      enabledForFailure: true,
      tools: [spotBugs(pattern: 'target/spotbugsXml.xml')]
    )
  }
}
```

### Combined Static Analysis

Run all static analysis tools:

```groovy
post {
  always {
    recordIssues(
      enabledForFailure: true,
      tools: [
        checkStyle(pattern: 'target/checkstyle-result.xml'),
        pmdParser(pattern: 'target/pmd.xml'),
        spotBugs(pattern: 'target/spotbugsXml.xml')
      ],
      unstableTotalAll: 100,  // Mark unstable if total > 100 issues
      failedTotalAll: 1000    // Fail if total > 1000 issues
    )
  }
}
```

## HTML Report Publishing

### publishHTML Step

Publish any HTML report directory to Jenkins:

```groovy
post {
  always {
    publishHTML(target: [
      allowMissing: false,
      alwaysLinkToLastBuild: true,
      keepAll: true,
      reportDir: 'target/site',           // Directory containing index.html
      reportFiles: 'index.html',          // Start page
      reportName: 'Maven Report'          // Display name in Jenkins
    ])
  }
}
```

### Multiple Reports

Publish several report types:

```groovy
post {
  always {
    publishHTML(target: [
      reportDir: 'target/site/jacoco',
      reportFiles: 'index.html',
      reportName: 'JaCoCo Coverage'
    ])
    publishHTML(target: [
      reportDir: 'target/site',
      reportFiles: 'index.html',
      reportName: 'Maven Surefire Report'
    ])
    publishHTML(target: [
      reportDir: 'coverage',
      reportFiles: 'index.html',
      reportName: 'Code Coverage Report'
    ])
  }
}
```

## Email Notifications

### Basic Email

Send email on build failure:

```groovy
post {
  failure {
    emailext(
      subject: 'Build Failed: ${JOB_NAME} ${BUILD_NUMBER}',
      body: '''
        Build Details:
        Job: ${JOB_NAME}
        Build: ${BUILD_NUMBER}
        Status: ${BUILD_STATUS}
        Commit: ${GIT_COMMIT}

        See details: ${BUILD_URL}console
      ''',
      to: 'team@example.com'
    )
  }
}
```

### Email with Attachments

```groovy
post {
  failure {
    emailext(
      subject: 'Build Failed: ${JOB_NAME}',
      body: '${BUILD_LOG_EXCERPT}',
      to: '${BUILD_USER_EMAIL}',
      mimeType: 'text/html',
      attachmentsPattern: '**/test-results.xml,coverage/**'
    )
  }
}
```

### Conditional Email

```groovy
post {
  always {
    script {
      if (currentBuild.result == 'FAILURE') {
        emailext(
          subject: 'FAILED: ${JOB_NAME}',
          body: 'Build failed. See ${BUILD_URL}',
          to: 'oncall@example.com,${BUILD_USER_EMAIL}'
        )
      } else if (currentBuild.result == 'UNSTABLE') {
        emailext(
          subject: 'UNSTABLE: ${JOB_NAME}',
          body: 'Tests failed (build passed). See ${BUILD_URL}',
          to: 'team@example.com'
        )
      }
    }
  }
}
```

## Complete Example: Full Test & Reporting Pipeline

```groovy
pipeline {
  agent any

  options {
    timestamps()
    buildDiscarder(logRotator(numToKeepStr: '30'))
  }

  stages {
    stage('Checkout') {
      steps {
        checkout scm
      }
    }

    stage('Build') {
      steps {
        sh 'mvn clean compile'
      }
    }

    stage('Unit Tests') {
      steps {
        sh 'mvn test'
      }
    }

    stage('Integration Tests') {
      steps {
        sh 'mvn verify'
      }
    }

    stage('Code Coverage') {
      steps {
        sh 'mvn jacoco:report'
      }
    }

    stage('Static Analysis') {
      parallel {
        stage('Checkstyle') {
          steps {
            sh 'mvn checkstyle:checkstyle'
          }
        }
        stage('PMD') {
          steps {
            sh 'mvn pmd:pmd'
          }
        }
        stage('SpotBugs') {
          steps {
            sh 'mvn spotbugs:spotbugs'
          }
        }
      }
    }

    stage('SonarQube Analysis') {
      steps {
        sh '''
          mvn sonar:sonar \
            -Dsonar.projectKey=my-app \
            -Dsonar.sources=src \
            -Dsonar.host.url=${SONAR_HOST_URL} \
            -Dsonar.login=${SONAR_LOGIN}
        '''
      }
    }

    stage('Package') {
      steps {
        sh 'mvn package -DskipTests'
      }
    }
  }

  post {
    always {
      // Test Results
      junit 'target/surefire-reports/**/*.xml'

      // Code Coverage
      publishHTML(target: [
        reportDir: 'target/site/jacoco',
        reportFiles: 'index.html',
        reportName: 'JaCoCo Coverage'
      ])
      step([$class: 'JacocoPublisher'])

      // Static Analysis
      recordIssues(
        enabledForFailure: true,
        tools: [
          checkStyle(pattern: 'target/checkstyle-result.xml'),
          pmdParser(pattern: 'target/pmd.xml'),
          spotBugs(pattern: 'target/spotbugsXml.xml')
        ]
      )

      // Artifacts
      archiveArtifacts artifacts: 'target/*.jar', allowEmptyArchive: false
    }

    success {
      echo 'Build successful'
      archiveArtifacts artifacts: 'target/**', fingerprint: true
    }

    failure {
      emailext(
        subject: 'Build Failed: ${JOB_NAME}',
        body: '''
          Build: ${BUILD_NUMBER}
          Status: FAILED

          Test Results: ${BUILD_URL}testReport
          Console: ${BUILD_URL}console
        ''',
        to: 'team@example.com'
      )
    }

    unstable {
      echo 'Build is unstable (test failures)'
      emailext(
        subject: 'Build Unstable: ${JOB_NAME}',
        body: 'Tests failed. See ${BUILD_URL}testReport',
        to: 'team@example.com'
      )
    }
  }
}
```
